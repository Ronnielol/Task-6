def initialize(file)
  @movies = CSV.foreach(
    file, col_sep: '|', headers: HEADERS,
          force_quotes: 'false', converters: [:numeric]
  ).map { |row| Cinema::Movie.create(get_movie_attrs(row)) }
end

def stats(arg)
  @movies
    .map(&arg)
    .compact
    .each_with_object(Hash.new(0)) { |o, h| h[o] += 1 }
end
